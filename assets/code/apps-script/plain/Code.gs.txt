   IMAGEN / RECURSOS
====================================================== */

function getTrophyImageDataUrl_() {
  try {
    const file = DriveApp.getFileById(TROPHY_IMAGE_ID);
    const blob = file.getBlob();
    const contentType = blob.getContentType();
    const base64 = Utilities.base64Encode(blob.getBytes());

    return `data:${contentType};base64,${base64}`;
  } catch (error) {
    Logger.log('Error cargando imagen de copa: ' + error.message);
    return '';
  }
}

/* =====================================================
   LECTURA DE GOOGLE SHEETS
====================================================== */

function sheetToObjectsById_(spreadsheetId) {
  return sheetToObjects_(spreadsheetId, null);
}

function sheetToObjectsByName_(spreadsheetId, sheetName) {
  return sheetToObjects_(spreadsheetId, sheetName);
}

function sheetToObjects_(spreadsheetId, sheetName) {
  const ss = SpreadsheetApp.openById(spreadsheetId);

  let sheet;

  if (sheetName) {
    sheet = ss.getSheetByName(sheetName);
  } else {
    sheet = ss.getSheets()[0];
  }

  if (!sheet) {
    throw new Error(
      sheetName
        ? `No se encontró la hoja "${sheetName}" en el archivo: ${spreadsheetId}`
        : `No se encontró ninguna hoja en el archivo: ${spreadsheetId}`
    );
  }

  const values = sheet.getDataRange().getDisplayValues();

  if (!values || values.length < 2) {
    return [];
  }

  const headers = values[0].map(header => String(header || '').trim());

  return values.slice(1)
    .filter(row => row.some(cell => String(cell || '').trim() !== ''))
    .map(row => {
      const obj = {};

      headers.forEach((header, index) => {
        if (!header) return;
        obj[header] = row[index];
      });

      return obj;
    });
}

function loadTable_(tableKey) {
  const tableConfig = SPREADSHEETS[tableKey];

  if (!tableConfig) {
    throw new Error(`No existe configuración para la tabla histórica: ${tableKey}`);
  }

  return sheetToObjectsById_(tableConfig.id);
}

function loadPredictionTable_(tableKey) {
  const tableConfig = PREDICCION_SHEETS[tableKey];

  if (!tableConfig) {
    throw new Error(`No existe configuración para la tabla de predicción: ${tableKey}`);
  }

  return sheetToObjectsByName_(tableConfig.id, tableConfig.sheetName);
}

function loadPredictionTableOptional_(tableKey) {
  try {
    return loadPredictionTable_(tableKey);
  } catch (error) {
    Logger.log('Tabla de predicción opcional no cargada [' + tableKey + ']: ' + error.message);
    return [];
  }
}

/* =====================================================
   FILTROS GLOBALES
====================================================== */

function buildFilters_(selecciones, partidos, jugadores, eventos) {
  selecciones = selecciones || [];
  partidos = partidos || [];
  jugadores = jugadores || [];
  eventos = eventos || [];

  const anios = uniqueSortedNumbers_([
    ...getValues_(selecciones, ['anio', 'year']),
    ...getValues_(partidos, ['anio', 'year']),
    ...getValues_(jugadores, ['anio', 'year']),
    ...getValues_(eventos, ['anio', 'year'])
  ]);

  const seleccionesLista = uniqueSortedText_([
    ...getValues_(selecciones, ['seleccion', 'team_name', 'equipo']),
    ...getValues_(partidos, ['seleccion', 'team_name', 'equipo', 'home_team_name', 'away_team_name']),
    ...getValues_(jugadores, ['seleccion', 'team_name', 'equipo']),
    ...getValues_(eventos, ['seleccion', 'team_name', 'equipo'])
  ]);

  const confederaciones = uniqueSortedText_([
    ...getValues_(selecciones, ['confederation_code', 'confederacion'])
  ]);

  const fases = uniqueSortedText_([
    ...getValues_(selecciones, ['fase_alcanzada', 'fase', 'stage_reached', 'stage', 'round']),
    ...getValues_(partidos, ['fase', 'stage', 'round']),
    ...getValues_(eventos, ['fase', 'stage', 'round'])
  ]);

  return {
    anios,
    selecciones: seleccionesLista,
    confederaciones,
    fases
  };
}

function getValues_(rows, fieldNames) {
  const values = [];

  rows.forEach(row => {
    fieldNames.forEach(fieldName => {
      const value = row[fieldName];

      if (value !== '' && value !== null && value !== undefined) {
        values.push(value);
      }
    });
  });

  return values;
}

function uniqueSortedText_(values) {
  return [...new Set(
    values
      .map(value => String(value || '').trim())
      .filter(value => value !== '')
  )].sort((a, b) => a.localeCompare(b));
}

function uniqueSortedNumbers_(values) {
  return [...new Set(
    values
      .map(value => Number(value))
      .filter(value => !isNaN(value))
  )].sort((a, b) => a - b);
}
